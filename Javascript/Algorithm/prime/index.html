function primeFactors(n){
  const result = new Set()
  // 2로 나뉘지 않을 때까지 나눠주기
  while (n%2 === 0) {
    if (result.at(-1) !== 2){
      result.add(2)
    }
    n = n/2
  }
  // n은 이제 홀수이다. 홀수만 확인하기 위해서 2씩 증가시킨다.
  // √n까지만 확인하면 된다.
  for (let i=3; i*i<=n; i=i+2){
    while (n%i === 0){
      result.add(i)
      n = n/i
    }
  }
  // 마지막 n이 2보다 큰 소수인 경우, result에 추가해준다.
  if (n>2){
    result.add(n)
  }
  return result
}
